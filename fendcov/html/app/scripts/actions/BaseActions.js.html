(function () {
  'use strict';
 
  var AppDispatcher = require('../dispatcher/AppDispatcher');
  var request = require('superagent');
 
  module.exports = {
    get: function get(url, actionType, token) {
      request.get(url).set('x-access-token', token).end(function (err, result) {
        AppDispatcher.dispatch({
          actionType: actionType,
          data: result.body
        });
      });
    },
 
    put: function put(url, data, actionType, token) {
      request.put(url).set('x-access-token', token).send(data).end(function (err, result) {
        AppDispatcher.dispatch({
          actionType: actionType,
          data: result.body
        });
      });
    },
 
    delete: function _delete(url, actionType, token) {
      request.delete(url).set('x-access-token', token).end(function (err, result) {
        AppDispatcher.dispatch({
          actionType: actionType,
          data: result.body
        });
      });
    },
 
    post: function post(url, data, actionType, token) {
      request.post(url).set('x-access-token', token).send(data).end(function (err, result) {
        AppDispatcher.dispatch({
          actionType: actionType,
          data: result.body
        });
      });
    }
  };
})();
